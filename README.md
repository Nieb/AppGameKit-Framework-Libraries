### Common Abbreviations:

    Pos = "Position"
    Rot = "Rotation"
    Siz = "Size"
    Scl = "Scale"
    Nrm = "Normal"
    Rds = "Radius"
    Dir = "Direction" or "Directory"
    Clr = "Color"

    Dlt = "Delta"

    Deg = "Degrees"
    Rad = "Radian"

    Inv = "Invert"
    Cmp = "Complement"
    Rcp = "Reciprocal"
